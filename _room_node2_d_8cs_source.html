338 public void SetCellActivity(int row, int column, CellActivity activity)
339 {
340 if (!CellIndexExists(row, column))
341 throw new IndexOutOfRangeException($"Cell index does not exist: ({row}, {column}).");
342
343 switch (activity)
344 {
345 case CellActivity.None:
346 break;
347 case CellActivity.Activate:
348 ActiveCells[row][column] = true;
349 break;
350 case CellActivity.Deactivate:
351 ActiveCells[row][column] = false;
352 break;
353 case CellActivity.Toggle:
354 ActiveCells[row][column] = !ActiveCells[row][column];
355 break;
356 default:
357 throw new NotImplementedException($"Unhandled cell activity: {activity}.");
358 }
359 }

380 public bool SetCellActivities(Vector2I startIndex, Vector2I endIndex, CellActivity activity)
381 {
382 if (activity == CellActivity.None || !CellIndexRangeExists(startIndex, endIndex))
383 return false;
384
385 var startRow = Mathf.Min(startIndex.X, endIndex.X);
386 var endRow = Mathf.Max(startIndex.X, endIndex.X);
387 var startColumn = Mathf.Min(startIndex.Y, endIndex.Y);
388 var endColumn = Mathf.Max(startIndex.Y, endIndex.Y);
389
390 for (int i = startRow; i <= endRow; i++)
391 {
392 for (int j = startColumn; j <= endColumn; j++)
393 {
394 SetCellActivity(i, j, activity);
395 }
396 }
397
398 return true;
399 }

405 public void SizeActiveCells()
406 {
407 while (ActiveCells.Count > Rows)
408 {
409 ActiveCells.RemoveAt(ActiveCells.Count - 1);
410 }
411
412 foreach (var row in ActiveCells)
413 {
414 while (row.Count > Columns)
415 {
416 row.RemoveAt(row.Count - 1);
417 }
418
419 while (row.Count < Columns)
420 {
421 row.Add(true);
422 }
423 }
424
425 while (ActiveCells.Count < Rows)
426 {
427 ActiveCells.Add(new Godot.Collections.Array<bool>(Enumerable.Repeat(true, Columns)));
428 }
429 }

435 public Vector2I FindClosestActiveCellIndex(Vector2 position)
436 {
437 var fastIndex = GlobalPositionToCellIndex(position);
438
439 if (CellIndexExists(fastIndex.X, fastIndex.Y) && ActiveCells[fastIndex.X][fastIndex.Y])
440 return fastIndex;
441
442 var index = Vector2I.Zero;
443 var minDistance = float.PositiveInfinity;
444
445 for (int i = 0; i < ActiveCells.Count; i++)
446 {
447 var row = ActiveCells[i];
448
449 for (int j = 0; j < row.Count; j++)
450 {
451 if (row[j])
452 {
453 var delta = CellCenterGlobalPosition(i, j) - position;
454 var distance = delta.LengthSquared();
455
456 if (distance < minDistance)
457 {
458 minDistance = distance;
459 index = new Vector2I(i, j);
460 }
461 }
462 }
463 }
464
465 return index;
466 }

475 public DoorDirection FindClosestDoorDirection(int row, int column, Vector2 position)
476 {
477 Span<DoorDirection> directions = stackalloc DoorDirection[]
478 {
479 DoorDirection.North,
480 DoorDirection.East,
481 DoorDirection.South,
482 DoorDirection.West,
483 };
484
485 Span<Vector2> vectors = stackalloc Vector2[]
486 {
487 new Vector2(0, -1),
488 new Vector2(1, 0),
489 new Vector2(0, 1),
490 new Vector2(-1, 0),
491 };
492
493 var index = 0;
494 var maxDistance = float.NegativeInfinity;
495 var delta = position - CellCenterGlobalPosition(row, column);
496
497 for (int i = 0; i < vectors.Length; i++)
498 {
499 var distance = delta.Dot(vectors[i]);
500
501 if (distance > maxDistance)
502 {
503 maxDistance = distance;
504 index = i;
505 }
506 }
507
508 return directions[index];
509 }
