267 public Vector2I FindClosestActiveCellIndex(Vector2 position)
268 {
269 var fastIndex = GlobalPositionToCellIndex(position);
270
271 if (this.CellIndexExists(fastIndex.X, fastIndex.Y) && ActiveCells[fastIndex.X][fastIndex.Y])
272 return fastIndex;
273
274 var index = Vector2I.Zero;
275 var minDistance = float.PositiveInfinity;
276
277 for (int i = 0; i < ActiveCells.Count; i++)
278 {
279 var row = ActiveCells[i];
280
281 for (int j = 0; j < row.Count; j++)
282 {
283 if (row[j])
284 {
285 var delta = CellCenterGlobalPosition(i, j) - position;
286 var distance = delta.LengthSquared();
287
288 if (distance < minDistance)
289 {
290 minDistance = distance;
291 index = new Vector2I(i, j);
292 }
293 }
294 }
295 }
296
297 return index;
298 }

307 public DoorDirection FindClosestDoorDirection(int row, int column, Vector2 position)
308 {
309 Span<DoorDirection> directions = stackalloc DoorDirection[]
310 {
311 DoorDirection.North,
312 DoorDirection.East,
313 DoorDirection.South,
314 DoorDirection.West,
315 };
316
317 Span<Vector2> vectors = stackalloc Vector2[]
318 {
319 new Vector2(0, -1),
320 new Vector2(1, 0),
321 new Vector2(0, 1),
322 new Vector2(-1, 0),
323 };
324
325 var index = 0;
326 var maxDistance = float.NegativeInfinity;
327 var delta = position - CellCenterGlobalPosition(row, column);
328
329 for (int i = 0; i < vectors.Length; i++)
330 {
331 var distance = delta.Dot(vectors[i]);
332
333 if (distance > maxDistance)
334 {
335 maxDistance = distance;
336 index = i;
337 }
338 }
339
340 return directions[index];
341 }
