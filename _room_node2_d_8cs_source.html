260 public Vector2I FindClosestActiveCellIndex(Vector2 position)
261 {
262 var fastIndex = GlobalPositionToCellIndex(position);
263
264 if (this.CellIndexExists(fastIndex.X, fastIndex.Y) && ActiveCells[fastIndex.X][fastIndex.Y])
265 return fastIndex;
266
267 var index = Vector2I.Zero;
268 var minDistance = float.PositiveInfinity;
269
270 for (int i = 0; i < ActiveCells.Count; i++)
271 {
272 var row = ActiveCells[i];
273
274 for (int j = 0; j < row.Count; j++)
275 {
276 if (row[j])
277 {
278 var delta = CellCenterGlobalPosition(i, j) - position;
279 var distance = delta.LengthSquared();
280
281 if (distance < minDistance)
282 {
283 minDistance = distance;
284 index = new Vector2I(i, j);
285 }
286 }
287 }
288 }
289
290 return index;
291 }

300 public DoorDirection FindClosestDoorDirection(int row, int column, Vector2 position)
301 {
302 Span<DoorDirection> directions = stackalloc DoorDirection[]
303 {
304 DoorDirection.North,
305 DoorDirection.East,
306 DoorDirection.South,
307 DoorDirection.West,
308 };
309
310 Span<Vector2> vectors = stackalloc Vector2[]
311 {
312 new Vector2(0, -1),
313 new Vector2(1, 0),
314 new Vector2(0, 1),
315 new Vector2(-1, 0),
316 };
317
318 var index = 0;
319 var maxDistance = float.NegativeInfinity;
320 var delta = (position - CellCenterGlobalPosition(row, column)) / CellSize;
321
322 for (int i = 0; i < vectors.Length; i++)
323 {
324 var distance = delta.Dot(vectors[i]);
325
326 if (distance > maxDistance)
327 {
328 maxDistance = distance;
329 index = i;
330 }
331 }
332
333 return directions[index];
334 }
