321 public void SetCellActivity(int row, int column, CellActivity activity)
322 {
323 if (!CellIndexExists(row, column))
324 throw new IndexOutOfRangeException($"Cell index does not exist: ({row}, {column}).");
325
326 switch (activity)
327 {
328 case CellActivity.None:
329 break;
330 case CellActivity.Activate:
331 ActiveCells[row][column] = true;
332 break;
333 case CellActivity.Deactivate:
334 ActiveCells[row][column] = false;
335 break;
336 case CellActivity.Toggle:
337 ActiveCells[row][column] = !ActiveCells[row][column];
338 break;
339 default:
340 throw new NotImplementedException($"Unhandled cell activity: {activity}.");
341 }
342 }

363 public bool SetCellActivities(Vector2I startIndex, Vector2I endIndex, CellActivity activity)
364 {
365 if (activity == CellActivity.None || !CellIndexRangeExists(startIndex, endIndex))
366 return false;
367
368 var startRow = Mathf.Min(startIndex.X, endIndex.X);
369 var endRow = Mathf.Max(startIndex.X, endIndex.X);
370 var startColumn = Mathf.Min(startIndex.Y, endIndex.Y);
371 var endColumn = Mathf.Max(startIndex.Y, endIndex.Y);
372
373 for (int i = startRow; i <= endRow; i++)
374 {
375 for (int j = startColumn; j <= endColumn; j++)
376 {
377 SetCellActivity(i, j, activity);
378 }
379 }
380
381 return true;
382 }

388 public void SizeActiveCells()
389 {
390 while (ActiveCells.Count > Rows)
391 {
392 ActiveCells.RemoveAt(ActiveCells.Count - 1);
393 }
394
395 foreach (var row in ActiveCells)
396 {
397 while (row.Count > Columns)
398 {
399 row.RemoveAt(row.Count - 1);
400 }
401
402 while (row.Count < Columns)
403 {
404 row.Add(true);
405 }
406 }
407
408 while (ActiveCells.Count < Rows)
409 {
410 ActiveCells.Add(new Godot.Collections.Array<bool>(Enumerable.Repeat(true, Columns)));
411 }
412 }

418 public Vector2I FindClosestActiveCellIndex(Vector2 position)
419 {
420 var fastIndex = GlobalPositionToCellIndex(position);
421
422 if (CellIndexExists(fastIndex.X, fastIndex.Y) && ActiveCells[fastIndex.X][fastIndex.Y])
423 return fastIndex;
424
425 var index = Vector2I.Zero;
426 var minDistance = float.PositiveInfinity;
427
428 for (int i = 0; i < ActiveCells.Count; i++)
429 {
430 var row = ActiveCells[i];
431
432 for (int j = 0; j < row.Count; j++)
433 {
434 if (row[j])
435 {
436 var delta = CellCenterGlobalPosition(i, j) - position;
437 var distance = delta.LengthSquared();
438
439 if (distance < minDistance)
440 {
441 minDistance = distance;
442 index = new Vector2I(i, j);
443 }
444 }
445 }
446 }
447
448 return index;
449 }

458 public DoorDirection FindClosestDoorDirection(int row, int column, Vector2 position)
459 {
460 Span<DoorDirection> directions = stackalloc DoorDirection[]
461 {
462 DoorDirection.North,
463 DoorDirection.East,
464 DoorDirection.South,
465 DoorDirection.West,
466 };
467
468 Span<Vector2> vectors = stackalloc Vector2[]
469 {
470 new Vector2(0, -1),
471 new Vector2(1, 0),
472 new Vector2(0, 1),
473 new Vector2(-1, 0),
474 };
475
476 var index = 0;
477 var maxDistance = float.NegativeInfinity;
478 var delta = position - CellCenterGlobalPosition(row, column);
479
480 for (int i = 0; i < vectors.Length; i++)
481 {
482 var distance = delta.Dot(vectors[i]);
483
484 if (distance > maxDistance)
485 {
486 maxDistance = distance;
487 index = i;
488 }
489 }
490
491 return directions[index];
492 }
