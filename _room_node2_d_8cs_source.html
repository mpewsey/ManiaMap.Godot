347 public void SetCellActivity(int row, int column, CellActivity activity)
348 {
349 if (!CellIndexExists(row, column))
350 throw new IndexOutOfRangeException($"Cell index does not exist: ({row}, {column}).");
351
352 switch (activity)
353 {
354 case CellActivity.None:
355 break;
356 case CellActivity.Activate:
357 ActiveCells[row][column] = true;
358 break;
359 case CellActivity.Deactivate:
360 ActiveCells[row][column] = false;
361 break;
362 case CellActivity.Toggle:
363 ActiveCells[row][column] = !ActiveCells[row][column];
364 break;
365 default:
366 throw new NotImplementedException($"Unhandled cell activity: {activity}.");
367 }
368 }

389 public bool SetCellActivities(Vector2I startIndex, Vector2I endIndex, CellActivity activity)
390 {
391 if (activity == CellActivity.None || !CellIndexRangeExists(startIndex, endIndex))
392 return false;
393
394 var startRow = Mathf.Min(startIndex.X, endIndex.X);
395 var endRow = Mathf.Max(startIndex.X, endIndex.X);
396 var startColumn = Mathf.Min(startIndex.Y, endIndex.Y);
397 var endColumn = Mathf.Max(startIndex.Y, endIndex.Y);
398
399 for (int i = startRow; i <= endRow; i++)
400 {
401 for (int j = startColumn; j <= endColumn; j++)
402 {
403 SetCellActivity(i, j, activity);
404 }
405 }
406
407 return true;
408 }

414 public void SizeActiveCells()
415 {
416 while (ActiveCells.Count > Rows)
417 {
418 ActiveCells.RemoveAt(ActiveCells.Count - 1);
419 }
420
421 foreach (var row in ActiveCells)
422 {
423 while (row.Count > Columns)
424 {
425 row.RemoveAt(row.Count - 1);
426 }
427
428 while (row.Count < Columns)
429 {
430 row.Add(true);
431 }
432 }
433
434 while (ActiveCells.Count < Rows)
435 {
436 ActiveCells.Add(new Godot.Collections.Array<bool>(Enumerable.Repeat(true, Columns)));
437 }
438 }

444 public Vector2I FindClosestActiveCellIndex(Vector2 position)
445 {
446 var fastIndex = GlobalPositionToCellIndex(position);
447
448 if (CellIndexExists(fastIndex.X, fastIndex.Y) && ActiveCells[fastIndex.X][fastIndex.Y])
449 return fastIndex;
450
451 var index = Vector2I.Zero;
452 var minDistance = float.PositiveInfinity;
453
454 for (int i = 0; i < ActiveCells.Count; i++)
455 {
456 var row = ActiveCells[i];
457
458 for (int j = 0; j < row.Count; j++)
459 {
460 if (row[j])
461 {
462 var delta = CellCenterGlobalPosition(i, j) - position;
463 var distance = delta.LengthSquared();
464
465 if (distance < minDistance)
466 {
467 minDistance = distance;
468 index = new Vector2I(i, j);
469 }
470 }
471 }
472 }
473
474 return index;
475 }

484 public DoorDirection FindClosestDoorDirection(int row, int column, Vector2 position)
485 {
486 Span<DoorDirection> directions = stackalloc DoorDirection[]
487 {
488 DoorDirection.North,
489 DoorDirection.East,
490 DoorDirection.South,
491 DoorDirection.West,
492 };
493
494 Span<Vector2> vectors = stackalloc Vector2[]
495 {
496 new Vector2(0, -1),
497 new Vector2(1, 0),
498 new Vector2(0, 1),
499 new Vector2(-1, 0),
500 };
501
502 var index = 0;
503 var maxDistance = float.NegativeInfinity;
504 var delta = position - CellCenterGlobalPosition(row, column);
505
506 for (int i = 0; i < vectors.Length; i++)
507 {
508 var distance = delta.Dot(vectors[i]);
509
510 if (distance > maxDistance)
511 {
512 maxDistance = distance;
513 index = i;
514 }
515 }
516
517 return directions[index];
518 }
