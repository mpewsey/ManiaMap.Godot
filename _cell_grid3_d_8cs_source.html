100 private void PopulateCells()
101 {
102 SizeCells();
103 var index = 0;
104 var cells = Room.ActiveCells;
105 var cellSize = Room.CellSize;
106 var displayCells = DisplayCells();
107 var edgeColor = ManiaMapProjectSettings.GetRoom3DCellLineColor();
108 var activeColor = ManiaMapProjectSettings.GetRoom3DActiveCellColor();
109 var inactivecolor = ManiaMapProjectSettings.GetRoom3DInactiveCellColor();
110
111 for (int i = 0; i < cells.Count; i++)
112 {
113 var row = cells[i];
114
115 for (int j = 0; j < row.Count; j++)
116 {
117 var cell = GetChild<MeshInstance3D>(index++);
118 var edges = cell.GetChild<MeshInstance3D>(0);
119 edges.SetInstanceShaderParameter(AlbedoParameterName, edgeColor);
120 var color = row[j] ? activeColor : inactivecolor;
121 cell.SetInstanceShaderParameter(AlbedoParameterName, color);
122 cell.Position = Room.CellCenterLocalPosition(i, j);
123 cell.Scale = cellSize;
124 cell.Visible = displayCells;
125 }
126 }
