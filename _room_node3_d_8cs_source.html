342 public Vector2I FindClosestActiveCellIndex(Vector3 position)
343 {
344 var fastIndex = GlobalPositionToCellIndex(position);
345
346 if (this.CellIndexExists(fastIndex.X, fastIndex.Y) && ActiveCells[fastIndex.X][fastIndex.Y])
347 return fastIndex;
348
349 var index = Vector2I.Zero;
350 var minDistance = float.PositiveInfinity;
351
352 for (int i = 0; i < ActiveCells.Count; i++)
353 {
354 var row = ActiveCells[i];
355
356 for (int j = 0; j < row.Count; j++)
357 {
358 if (row[j])
359 {
360 var delta = CellCenterGlobalPosition(i, j) - position;
361 var distance = new Vector2(delta.X, delta.Z).LengthSquared();
362
363 if (distance < minDistance)
364 {
365 minDistance = distance;
366 index = new Vector2I(i, j);
367 }
368 }
369 }
370 }
371
372 return index;
373 }

382 public DoorDirection FindClosestDoorDirection(int row, int column, Vector3 position)
383 {
384 Span<DoorDirection> directions = stackalloc DoorDirection[]
385 {
386 DoorDirection.North,
387 DoorDirection.East,
388 DoorDirection.South,
389 DoorDirection.West,
390 DoorDirection.Top,
391 DoorDirection.Bottom,
392 };
393
394 Span<Vector3> vectors = stackalloc Vector3[]
395 {
396 new Vector3(0, 0, -1),
397 new Vector3(1, 0, 0),
398 new Vector3(0, 0, 1),
399 new Vector3(-1, 0, 0),
400 new Vector3(0, 1, 0),
401 new Vector3(0, -1, 0),
402 };
403
404 var index = 0;
405 var maxDistance = float.NegativeInfinity;
406 var delta = (position - CellCenterGlobalPosition(row, column)) / CellSize;
407
408 for (int i = 0; i < vectors.Length; i++)
409 {
410 var distance = delta.Dot(vectors[i]);
411
412 if (distance > maxDistance)
413 {
414 maxDistance = distance;
415 index = i;
416 }
417 }
418
419 return directions[index];
420 }
