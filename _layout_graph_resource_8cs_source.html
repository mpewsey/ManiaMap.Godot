248 public LayoutGraphEdge AddEdge(int fromNode, int toNode)
249 {
250 if (!Nodes.ContainsKey(fromNode))
251 throw new ArgumentException($"{nameof(fromNode)} does not exist: {fromNode}.");
252 if (!Nodes.ContainsKey(toNode))
253 throw new ArgumentException($"{nameof(toNode)} does not exist: {toNode}.");
254
255 if (!ContainsEdge(fromNode, toNode))
256 {
257 var edge = new LayoutGraphEdge(fromNode, toNode);
258 Edges.Add(new Vector2I(fromNode, toNode), edge);
259 edge.Changed += OnSubresourceChanged;
260 SetDirty();
261 return edge;
262 }
263
264 return null;
265 }
