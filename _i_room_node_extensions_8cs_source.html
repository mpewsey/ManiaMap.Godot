59 public static bool SetCellActivities(this IRoomNode room, Vector2I startIndex, Vector2I endIndex, CellActivity activity)
60 {
61 if (activity == CellActivity.None || !room.CellIndexRangeExists(startIndex, endIndex))
62 return false;
63
64 var startRow = Mathf.Min(startIndex.X, endIndex.X);
65 var endRow = Mathf.Max(startIndex.X, endIndex.X);
66 var startColumn = Mathf.Min(startIndex.Y, endIndex.Y);
67 var endColumn = Mathf.Max(startIndex.Y, endIndex.Y);
68
69 for (int i = startRow; i <= endRow; i++)
70 {
71 for (int j = startColumn; j <= endColumn; j++)
72 {
73 room.SetCellActivity(i, j, activity);
74 }
75 }
76
77 return true;
78 }

103 public static void SetCellActivity(this IRoomNode room, int row, int column, CellActivity activity)
104 {
105 if (!room.CellIndexExists(row, column))
106 throw new IndexOutOfRangeException($"Cell index does not exist: ({row}, {column}).");
107
108 switch (activity)
109 {
110 case CellActivity.None:
111 break;
112 case CellActivity.Activate:
113 room.ActiveCells[row][column] = true;
114 break;
115 case CellActivity.Deactivate:
116 room.ActiveCells[row][column] = false;
117 break;
118 case CellActivity.Toggle:
119 room.ActiveCells[row][column] = !room.ActiveCells[row][column];
120 break;
121 default:
122 throw new NotImplementedException($"Unhandled cell activity: {activity}.");
123 }
124 }

130 public static void SizeActiveCells(this IRoomNode room)
131 {
132 var rows = room.Rows;
133 var columns = room.Columns;
134 var activeCells = room.ActiveCells;
135
136 while (activeCells.Count > rows)
137 {
138 activeCells.RemoveAt(activeCells.Count - 1);
139 }
140
141 foreach (var row in activeCells)
142 {
143 while (row.Count > columns)
144 {
145 row.RemoveAt(row.Count - 1);
146 }
147
148 while (row.Count < columns)
149 {
150 row.Add(true);
151 }
152 }
153
154 while (activeCells.Count < rows)
155 {
156 activeCells.Add(new Godot.Collections.Array<bool>(Enumerable.Repeat(true, columns)));
157 }
158 }

165 public static RoomTemplate GetMMRoomTemplate(this IRoomNode room, int id, string name)
166 {
167 var roomNode = (Node)room;

169 room.AddMMDoors(cells);

182 throw;

191 private static Array2D<Cell> GetMMCells(this IRoomNode room)
192 {
193 room.SizeActiveCells();
194 var rows = room.Rows;
195 var columns = room.Columns;
196 var activeCells = room.ActiveCells;
197 var cells = new Array2D<Cell>(rows, columns);
198
199 for (int i = 0; i < rows; i++)
200 {
201 var row = activeCells[i];
202
203 for (int j = 0; j < columns; j++)
204 {
205 if (row[j])
206 cells[i, j] = Cell.New;
207 }
208 }
209
210 return cells;

362 public static bool UpdateRoomTemplate(this IRoomNode room)
363 {
364 var roomNode = (Node)room;
365 var sceneFilePath = roomNode.SceneFilePath;
366
367 if (!SceneIsSavedToFile(sceneFilePath))
368 {
369 GD.PrintErr("Scene must be saved to file first.");
370 return false;
371 }
372
373 room.AutoAssign();
374 room.RoomTemplate.Initialize(room);
375
376 if (!ResourceIsSavedToFile(room.RoomTemplate, sceneFilePath))
377 {
378 var path = roomNode.SceneFilePath.GetBaseName() + ".room_template.tres";
379 ResourceSaver.Save(room.RoomTemplate, path);
380 room.RoomTemplate = ResourceLoader.Load<RoomTemplateResource>(path);
381 GD.Print($"Saved room template to: {path}");
382 }
383
384 if (Engine.IsEditorHint())
385 EditorInterface.Singleton.MarkSceneAsUnsaved();
386
387 GD.PrintRich("[color=#00ff00]Room template updated.[/color]");
388 return true;
389 }

412 public static void AutoAssign(this IRoomNode room)
413 {
414 var roomNode = (Node)room;
415 room.RoomTemplate ??= new RoomTemplateResource() { TemplateName = roomNode.Name };
416 room.RoomTemplate.Id = Rand.AutoAssignId(room.RoomTemplate.Id);
417 room.SizeActiveCells();
418 var nodes = roomNode.FindChildren("*", room.GetCellChildClassName(), true, false);
419
420 switch (room)
421 {
422 case RoomNode2D room2d:
423 AutoAssign2D(room2d, nodes);
424 break;
425 case RoomNode3D room3d:
426 AutoAssign3D(room3d, nodes);
427 break;
428 default:
429 throw new NotImplementedException($"Unhandled type: {room.GetType()}");
430 }
431
432 if (Engine.IsEditorHint())
433 EditorInterface.Singleton.MarkSceneAsUnsaved();
434
435 GD.PrintRich($"[color=#00ff00]Performed auto assign on {nodes.Count} cell children.[/color]");
436 }
