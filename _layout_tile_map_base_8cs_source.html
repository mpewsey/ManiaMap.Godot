123 protected void SetTiles(TileMap tileMap, int z)
124 {
125 tileMap.Clear();
126
127 foreach (var room in RoomsByLayer[z])
128 {
129 var cells = room.Template.Cells;
130 var roomState = LayoutState?.RoomStates[room.Id];
131
132 for (int i = 0; i < cells.Rows; i++)
133 {
134 for (int j = 0; j < cells.Columns; j++)
135 {
136 var cell = cells[i, j];
137
138 // If cell is empty, go to next cell.
139 if (cell == null)
140 continue;
141
142 var position = new Vector2DInt(i, j);
143 var isCompletelyVisible = roomState == null || roomState.CellIsVisible(position);
144
145 // If room state is defined and is not visible, go to next cell.
146 if (!isCompletelyVisible && !roomState.IsVisible)
147 continue;
148
149 var cellPosition = new Vector2I(room.Position.Y + j, room.Position.X + i);
150
151 // Get adjacent cells
152 var north = cells.GetOrDefault(i - 1, j);
153 var south = cells.GetOrDefault(i + 1, j);
154 var west = cells.GetOrDefault(i, j - 1);
155 var east = cells.GetOrDefault(i, j + 1);
156
157 // Get the wall or door tiles
158 var topTile = GetTileCoordinate(room, cell, null, position, DoorDirection.Top);
159 var bottomTile = GetTileCoordinate(room, cell, null, position, DoorDirection.Bottom);
160 var northTile = GetTileCoordinate(room, cell, north, position, DoorDirection.North);
161 var southTile = GetTileCoordinate(room, cell, south, position, DoorDirection.South);
162 var westTile = GetTileCoordinate(room, cell, west, position, DoorDirection.West);
163 var eastTile = GetTileCoordinate(room, cell, east, position, DoorDirection.East);
164
165 // Add cell background tile
166 var backgroundColor = isCompletelyVisible ? ColorUtility.ConvertColor4ToColor(room.Color) : RoomColor;
167 SetBackgroundTile(tileMap, cellPosition, backgroundColor);
168
169 // Draw tiles
170 SetTile(tileMap, TopLayer, cellPosition, topTile);
171 SetTile(tileMap, BottomLayer, cellPosition, bottomTile);
172 SetTile(tileMap, NorthLayer, cellPosition, northTile);
173 SetTile(tileMap, SouthLayer, cellPosition, southTile);
174 SetTile(tileMap, EastLayer, cellPosition, eastTile);
175 SetTile(tileMap, WestLayer, cellPosition, westTile);
176
177 // Draw features only if the cell is completely visible
178 if (isCompletelyVisible)
179 SetTile(tileMap, FeatureLayer, cellPosition, GetFeatureCoordinate(cell));
180 }
181 }
182 }

215 protected Vector2I GetFeatureCoordinate(Cell cell)
216 {
217 foreach (var feature in cell.Features)
218 {
219 var atlasPosition = MapTileSet.GetTileCoordinate(feature);
220
221 if (atlasPosition.X >= 0 && atlasPosition.Y >= 0)

236 protected Vector2I GetTileCoordinate(Room room, Cell cell, Cell neighbor, Vector2DInt position, DoorDirection direction)
237 {
238 if (Door.ShowDoor(DoorDrawMode, direction) && cell.GetDoor(direction) != null && DoorExists(room, position, direction))
239 return MapTileSet.GetTileCoordinate(MapTileType.GetDoorTileType(direction));
240
241 if (neighbor == null)
242 return MapTileSet.GetTileCoordinate(MapTileType.GetWallTileType(direction));
243
244 return new Vector2I(-1, -1);
245 }

253 protected bool DoorExists(Room room, Vector2DInt position, DoorDirection direction)
254 {
255 if (RoomDoors.TryGetValue(room.Id, out var doors))
256 {
257 foreach (var door in doors)

259 if (door.Matches(position, direction))
260 return true;
261 }
262 }
263
264 return false;
