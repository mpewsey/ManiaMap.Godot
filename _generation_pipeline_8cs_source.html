200 public async Task<PipelineResults> RunAttemptsAsync(int seed, int attempts = 10, int timeout = 5000, Dictionary<string, object> manualInputs = null, Action<string> logger = null)
201 {
202 manualInputs ??= new Dictionary<string, object>();
203
204 for (int i = 0; i < attempts; i++)
205 {
206 logger?.Invoke($"[Generation Pipeline] Beginning attempt {i + 1} / {attempts}...");
207 var inputs = new Dictionary<string, object>(manualInputs);
208 inputs.Add("RandomSeed", new RandomSeed(seed + i * 1447));
209 var token = new CancellationTokenSource(timeout).Token;
210 var results = await RunAsync(inputs, logger, token);
211
212 if (results.Success)
213 {
214 logger?.Invoke("[Generation Pipeline] Attempt successful.");
215 return results;
216 }
217 }
218
219 logger?.Invoke("[Generation Pipeline] Generation failed for all attempts.");
220 return new PipelineResults(manualInputs);
221 }

283 private void ValidateInputs(HashSet<string> names)
284 {
285 var nodes = FindInputNodes();
286
287 foreach (var node in nodes)
288 {
289 foreach (var name in node.InputNames())
290 {
291 if (!names.Add(name))
292 throw new DuplicateInputException($"Duplicate input name: (Name = {name}, NodePath = {GetPathTo(node)})");
293 }
294 }
295 }
296
302 private void ValidateSteps(HashSet<string> names)
303 {
304 var nodes = FindStepNodes();
305
306 foreach (var node in nodes)
307 {
308 foreach (var name in node.RequiredInputNames())
309 {
310 if (!names.Contains(name))
311 throw new MissingInputException($"Missing input name: (Name = {name}, NodePath = {GetPathTo(node)})");
312 }
313
314 foreach (var name in node.OutputNames())
315 {
316 names.Add(name);
317 }
318 }
319 }

327 public void SetRandomSeed(int seed)
328 {
329 var children = FindChildren("*", nameof(RandomSeedInput), true, false);
330
331 if (children.Count == 0)
332 throw new MissingInputException("Random seed input not found.");
333 if (children.Count > 1)
334 throw new DuplicateInputException($"Generation pipeline has multiple random seed inputs: {children.Count}.");
335
336 foreach (var child in children)
337 {
338 var input = (RandomSeedInput)child;
339 input.RandomSeed = seed;
340 }
341 }
