193 public async Task<PipelineResults> RunAttemptsAsync(int seed, int attempts = 10, int timeout = 5000, Dictionary<string, object> manualInputs = null, Action<string> logger = null)
194 {
195 manualInputs ??= new Dictionary<string, object>();
196
197 for (int i = 0; i < attempts; i++)
198 {
199 logger?.Invoke($"[Generation Pipeline] Beginning attempt {i + 1} / {attempts}...");
200 var inputs = new Dictionary<string, object>(manualInputs);
201 inputs.Add("RandomSeed", new RandomSeed(seed + i * 1447));
202 var token = new CancellationTokenSource(timeout).Token;
203 var results = await RunAsync(inputs, logger, token);
204
205 if (results.Success)
206 {
207 logger?.Invoke("[Generation Pipeline] Attempt successful.");
208 return results;
209 }
210 }
211
212 logger?.Invoke("[Generation Pipeline] Generation failed for all attempts.");
213 return new PipelineResults(manualInputs);
214 }

276 private void ValidateInputs(HashSet<string> names)
277 {
278 var nodes = FindInputNodes();
279
280 foreach (var node in nodes)
281 {
282 foreach (var name in node.InputNames())
283 {
284 if (!names.Add(name))
285 throw new DuplicateInputException($"Duplicate input name: (Name = {name}, NodePath = {GetPathTo(node)})");
286 }
287 }
288 }

295 private void ValidateSteps(HashSet<string> names)
296 {
297 var nodes = FindStepNodes();
298
299 foreach (var node in nodes)
300 {
301 foreach (var name in node.RequiredInputNames())
302 {
303 if (!names.Contains(name))
304 throw new MissingInputException($"Missing input name: (Name = {name}, NodePath = {GetPathTo(node)})");
305 }
306
307 foreach (var name in node.OutputNames())
308 {
309 names.Add(name);
310 }
311 }

320 public void SetRandomSeed(int seed)
321 {
322 var children = FindChildren("*", nameof(RandomSeedInput), true, false);
323
324 if (children.Count == 0)
325 throw new MissingInputException("Random seed input not found.");
326 if (children.Count > 1)
327 throw new DuplicateInputException($"Generation pipeline has multiple random seed inputs: {children.Count}.");
328
329 foreach (var child in children)
330 {
331 var input = (RandomSeedInput)child;
332 input.RandomSeed = seed;
333 }
