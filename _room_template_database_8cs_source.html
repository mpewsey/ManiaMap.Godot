113 private async Task<Dictionary<int, PackedScene>> LoadScenesAsync(IEnumerable<Room> rooms, bool useSubThreads = false)
114 {
115 var tasks = new Dictionary<int, Task<PackedScene>>();
116
117 foreach (var room in rooms)
118 {
119 var id = room.Template.Id;
120
121 if (!tasks.ContainsKey(id))
122 tasks.Add(id, GetRoomTemplate(id).LoadSceneAsync(useSubThreads));
123 }
124
125 await Task.WhenAll(tasks.Values);
126 var result = new Dictionary<int, PackedScene>(tasks.Count);
127
128 foreach (var pair in tasks)
129 {
130 result.Add(pair.Key, pair.Value.Result);
131 }
132
133 return result;
134 }

142 public async Task<List<RoomNode2D>> CreateRoom2DInstancesAsync(Node parent, int? z = null, bool useSubThreads = false)

144 var manager = ManiaMapManager.Current;
